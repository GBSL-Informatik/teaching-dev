import { useStore } from '@tdev-hooks/useStore'

# page-progress-state

Dieses Plugin ermöglicht eine Übersicht über alle speicherbaren Dokumente. So kann der Bearbeitungsstand in der Seitenleiste angezeigt werden.


export const Comp = observer(() => {
    const pageStore = useStore('pageStore');

    return (
        <div>
            <CodeBlock language="json">{JSON.stringify(pageStore.tree, null, 2)}</CodeBlock>;
        </div>
    );
});
