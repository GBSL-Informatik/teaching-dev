<TemplateCode>
    ```bash title="root@dokku"
    dokku apps:create {{app}}

    dokku postgres:create {{app}}
    dokku postgres:link {{app}} {{app}}

    dokku config:set --no-restart {{app}} BASE_URL=https://{{domain}}
    dokku domains:set {{app}} {{domain}}
    dokku ports:set {{app}} http:80:1337

    # create docker volumes for planka's persistent storage needs
    docker volume create {{app}}-user-avatars
    docker volume create {{app}}-project-background-images
    docker volume create {{app}}-attachments

    # mount the docker volumes
    dokku storage:mount {{app}} {{app}}-user-avatars:/app/public/user-avatars
    dokku storage:mount {{app}} {{app}}-project-background-images:/app/public/project-background-images
    dokku storage:mount {{app}} {{app}}-attachments:/app/private/attachments

    # edit env variables
    nano /home/dokku/{{app}}/ENV
    ```
</TemplateCode>

<TemplateCode>
```yaml title="backup_config.yml"
planka:
  files:
    - "/home/dokku/{{app}}/ENV"
  folders:
    - "/tmp/planka-backup/volumes"
  commands:
    volume-user-avatars:
      cmd: "docker run --rm --volumes-from {{app}}.web.1 -v /tmp/planka-backup/volumes:/backup ubuntu cp -r /app/public/user-avatars /backup/user-avatars"
      stage: "pre-backup"
    volume-project-background-images:
      cmd: "docker run --rm --volumes-from {{app}}.web.1 -v /tmp/planka-backup/volumes:/backup ubuntu cp -r /app/public/project-background-images /backup/project-background-images"
      stage: "pre-backup"
    volume-attachments:
      cmd: "docker run --rm --volumes-from {{app}}.web.1 -v /tmp/planka-backup/volumes:/backup ubuntu cp -r /app/private/attachments /backup/attachments"
      stage: "pre-backup"
    postgres:
      cmd: "dokku postgres:export {{app}}"
      to: "/database/{{app}}.dump"
      stage: backup
    cleanup:
      cmd: "rm -rf /tmp/planka-backup"
      stage: "post-backup"
```
</TemplateCode>
