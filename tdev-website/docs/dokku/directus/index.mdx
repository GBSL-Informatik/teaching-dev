## Dokku vorbereiten und deployen

<TemplateCode>
  ```bash title="root@dokku"
  # App erstellen
  dokku apps:create {{APP}}
  dokku redis:create {{APP}}
  dokku redis:link {{APP}} {{APP}}
  dokku postgres:create {{APP}} --image "postgis/postgis" --image-version "18-3.6"
  # dokku postgres:connect {{APP}}
  # run CREATE EXTENSION postgis;
  dokku postgres:link {{APP}} {{APP}}

  # Domain und Port setzen
  dokku domains:set {{APP}} {{DOMAIN}}
  dokku ports:set {{APP}} http:80:8055

  # Persistentes Verzeichnis für die SQLite-Datenbank mounten
  mkdir -p /var/lib/dokku/data/storage/{{APP}}/uploads
  mkdir /var/lib/dokku/data/storage/{{APP}}/extensions
  chown -R 1000:1000 /var/lib/dokku/data/storage/{{APP}}/uploads
  chown -R 1000:1000 /var/lib/dokku/data/storage/{{APP}}/extensions
  dokku storage:mount {{APP}} /var/lib/dokku/data/storage/{{APP}}/uploads:/directus/uploads
  dokku storage:mount {{APP}} /var/lib/dokku/data/storage/{{APP}}/extensions:/directus/extensions

  # GitHub OAuth und Drone-Konfiguration
  dokku config:set --no-restart {{APP}} DB_CLIENT="pg"
  dokku config:set --no-restart {{APP}} DB_CONNECTION_STRING="$(dokku postgres:info {{APP}} --dsn)"

  dokku config:set --no-restart {{APP}} CACHE_ENABLED="true"
  dokku config:set --no-restart {{APP}} CACHE_AUTO_PURGE="true"
  dokku config:set --no-restart {{APP}} CACHE_STORE="redis"
  dokku config:set --no-restart {{APP}} REDIS=$(dokku redis:info {{APP}} --dsn)

  dokku config:set --no-restart {{APP}} ADMIN_EMAIL="{{ADMIN_EMAIL}}"
  dokku config:set --no-restart {{APP}} ADMIN_PASSWORD="{{ADMIN_PASSWORD}}"
  dokku config:set --no-restart {{APP}} PUBLIC_URL="https://{{DOMAIN}}"
  dokku config:set --no-restart {{APP}} SECRET="{{SECRET}}"

  # Image deployen
  dokku git:from-image {{APP}} {{DIRECTUS_IMAGE}}

  # Optional: SSL aktivieren
  dokku letsencrypt:enable {{APP}}
  ```
</TemplateCode>
